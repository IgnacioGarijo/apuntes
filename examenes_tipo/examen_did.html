set.seed(20260511)
options(scipen = 999)

# Inspiracion: AIReF, Instrumentos financieros de apoyo a la pyme reforzados por el PRTR,
# con analisis causal contrafactual mediante diferencias en diferencias escalonado.

n <- 1200
emp <- data.frame(
  id_empresa = sprintf("P%05d", 1:n),
  ccaa = sample(c("Andalucia", "Cataluna", "Madrid", "Valencia", "Pais Vasco", "Castilla y Leon"), n, TRUE),
  sector = sample(c("Manufacturas", "Hosteleria", "Comercio", "TIC", "Servicios profesionales"), n, TRUE),
  tamano = sample(c("micro", "pequena", "mediana"), n, TRUE, c(.45,.40,.15)),
  empleados_base = round(exp(rnorm(n, log(18), .75))),
  endeudamiento_base = pmax(.05, rnorm(n, .52, .18))
)
cohortes <- sample(c(2020,2021,2022,2023,Inf), n, TRUE, c(.16,.18,.16,.10,.40))
emp$anio_apoyo <- cohortes
emp$empleados_base[sample(n,1)] <- 50000
emp$endeudamiento_base[sample(n,1)] <- 9

anios <- 2017:2024
panel <- merge(emp, data.frame(anio=anios))
panel$tratamiento <- as.integer(panel$anio >= panel$anio_apoyo & is.finite(panel$anio_apoyo))
panel$event_time <- ifelse(is.finite(panel$anio_apoyo), panel$anio - panel$anio_apoyo, NA)
size_eff <- ifelse(panel$tamano=="micro", -2, ifelse(panel$tamano=="pequena", 0, 4))
trend <- 1.2*(panel$anio-2017)
eff <- ifelse(panel$tratamiento==1, pmin(pmax(panel$event_time,0),4)*1.7 + 2.5, 0)
panel$empleo <- round(pmax(1, panel$empleados_base + trend + size_eff + eff + rnorm(nrow(panel),0,4)))
panel$credito_miles <- pmax(0, 80 + 6*panel$empleados_base + 15*(panel$anio-2017) +
                             ifelse(panel$tratamiento==1, 55 + 18*pmax(panel$event_time,0),0) +
                             rnorm(nrow(panel),0,75))
panel$tipo_interes <- pmax(.5, 4.5 - .08*(panel$anio-2017) - ifelse(panel$tratamiento==1, .25, 0) + rnorm(nrow(panel),0,.45))

if (!dir.exists("datos")) dir.create("datos")
write.csv(emp, "datos/empresas_pyme.csv", row.names=FALSE)
write.csv(panel, "datos/panel_pyme.csv", row.names=FALSE)
cat("Se han creado datos/empresas_pyme.csv y datos/panel_pyme.csv\n")

Existen algunas diferencias entre las covariables antes del tratamiento, siendo 2021 en general el año más dispar con respecto a los demás. Estas diferencias pueden poner en riesog la comparabilidad de los grupos, aunque en un modelo de diferencias en diferencias, si se cumplen las tendencias paralelas, aliviaría en cierta medida la preocupación por la comparabilidad de los grupos. Sin embargo, quedaría cierta preocupación por la diferencia en, por ejemplo, la composición por tamaño, puesto que la diferente respuesta por cohortes podría venir dada por esta variable u otras también con distribuciones dispares.

Podemos concluir que la política tuvo un efecto causal, produciendo una reducción de los tipos de interés y un aumento tanto del empleo como del crédito de las empresas. Los resultados son robustos para diferentes especificaciones (nevertreated vs not-yet-treated), y significativos para todas las cohortes, cumpliendo en todas las especificaciones la hipótesis de tendencias paralelas y sin mostrar signos de adelantamiento al tratamiento, contaminación u otros problemas que pueden afectar a este tipo de diseños cuasiexperimentales.
