set.seed(20260510)

# Inspiracion: AIReF, Spending Review de Beneficios Fiscales, que combina
# Diff-in-Diff, regresiones en discontinuidad y microsimuladores fiscales.
# El ejercicio simula una deduccion fiscal por I+D+i con un umbral de elegibilidad.

n <- 3500
empresas <- data.frame(
  id_empresa = sprintf("E%05d", 1:n),
  sector = sample(c("Industria", "TIC", "Servicios", "Comercio"), n, TRUE, c(.30,.22,.28,.20)),
  ccaa = sample(c("Andalucia", "Cataluna", "Madrid", "Valencia", "Pais Vasco", "Galicia"), n, TRUE),
  empleados_2020 = round(exp(rnorm(n, log(45), .85))),
  facturacion_2020_miles = round(exp(rnorm(n, log(3500), .9)), 1),
  intensidad_idi = pmax(0, pmin(.20, rnorm(n, .050, .035)))
)
# Outliers intencionados
empresas$empleados_2020[sample(n,1)] <- 99999
empresas$facturacion_2020_miles[sample(n,1)] <- -25

# Umbral de elegibilidad: intensidad de gasto en I+D+i >= 5% de la facturacion
empresas$running <- empresas$intensidad_idi - .05
empresas$elegible <- as.integer(empresas$running >= 0)

anios <- 2019:2023
panel <- merge(empresas, data.frame(anio = anios))
panel$post <- as.integer(panel$anio >= 2021)
panel$deduccion_miles <- ifelse(panel$elegible == 1 & panel$post == 1,
                                pmax(0, 25 + 1400*panel$running + rnorm(nrow(panel), 0, 18)), 0)
sector_eff <- setNames(c(5,12,0,-3), c("Industria","TIC","Servicios","Comercio"))
trend <- (panel$anio - 2019) * 3
trat_effect <- ifelse(panel$elegible == 1 & panel$post == 1, 9 + 2*(panel$anio-2021), 0)
panel$gasto_idi_miles <- 80 + 0.012*panel$facturacion_2020_miles +
  0.18*panel$empleados_2020 + sector_eff[panel$sector] + trend + trat_effect +
  45*panel$running + rnorm(nrow(panel), 0, 22)
panel$gasto_idi_miles <- pmax(0, panel$gasto_idi_miles)
panel$empleo <- round(pmax(1, panel$empleados_2020 + (panel$anio-2020)*rnorm(nrow(panel), 1.2, .6) +
                            ifelse(panel$elegible == 1 & panel$post == 1, 1.5, 0)))

if (!dir.exists("datos")) dir.create("datos")
write.csv(empresas, "datos/empresas_idi.csv", row.names = FALSE)
write.csv(panel, "datos/panel_idi.csv", row.names = FALSE)

cat("Se han creado datos/empresas_idi.csv y datos/panel_idi.csv\n")

1 Contexto

Este examen simula una evaluacion inspirada en el Spending Review de beneficios fiscales de AIReF. La politica evaluada es una deduccion fiscal para empresas con intensidad de gasto en I+D+i igual o superior al 5% de la facturacion. El objetivo es valorar si el umbral de elegibilidad genera un aumento causal del gasto en I+D+i.

Tiempo recomendado: 90 minutos. Se permite buscar documentacion en internet, pero no utilizar IA.
